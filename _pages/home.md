// =====================================================
// Utility: safe wrapper around gtag (if present)
// =====================================================
function safe_gtag(ev, params) {
  if (typeof gtag === "function") {
    gtag("event", ev, params);
  }
}

// ===============
// A. Scroll depth
// ===============
document.addEventListener("DOMContentLoaded", function () {
  const scrollPoints = [25, 50, 75, 100];
  const fired = new Set();

  function trackScroll() {
    const scrollTop = window.scrollY;
    const maxScroll = document.documentElement.scrollHeight - window.innerHeight;
    const pct = Math.floor((scrollTop / maxScroll) * 100);

    scrollPoints.forEach(function (p) {
      if (pct >= p && !fired.has(p)) {
        fired.add(p);

        safe_gtag("scroll_depth", {
          percent: p,
          event_category: "engagement",
          event_label: `Scrolled ${p}%`
        });
      }
    });
  }

  window.addEventListener("scroll", trackScroll);


  // ================================================
  // B. Dropdown & reading-time tracking per paper
  // ================================================
  const detailsList = document.querySelectorAll("details");

  detailsList.forEach(function (el) {
    const paperId = el.dataset.paperId || "unknown";

    el.addEventListener("toggle", function () {
      // Opening — start timing session
      if (el.open) {
        el.dataset.openedAt = Date.now();

        safe_gtag("dropdown_open", {
          paper_id: paperId,
          event_category: "interaction"
        });

      // Closing — stop timer
      } else if (!el.open && el.dataset.openedAt) {
        const duration = (Date.now() - el.dataset.openedAt) / 1000;

        safe_gtag("paper_read", {
          paper_id: paperId,
          duration_seconds: duration,
          event_category: "engagement"
        });

        el.dataset.openedAt = "";
      }
    });
  });

  // If user leaves the page without closing an opened detail
  document.addEventListener("visibilitychange", function () {
    if (document.visibilityState === "hidden") {
      detailsList.forEach(function (el) {
        if (el.open && el.dataset.openedAt) {
          const paperId = el.dataset.paperId || "unknown";
          const duration = (Date.now() - el.dataset.openedAt) / 1000;

          safe_gtag("paper_read", {
            paper_id: paperId,
            duration_seconds: duration,
            exited_without_closing: true,
            event_category: "engagement"
          });

          el.dataset.openedAt = "";
        }
      });
    }
  });



  // =====================================================
  // H. Track time spent on splash intro section
  //     - Start timer when section first enters viewport
  //     - Stop when user scrolls past it
  // =====================================================
  const splash = document.getElementById("splash-intro");
  let splashStart = null;
  let splashTimeSent = 0;
  let splashActive = false;

  function checkSplash() {
    const rect = splash.getBoundingClientRect();
    const inView = rect.bottom > 0 && rect.top < window.innerHeight;

    if (inView && !splashActive) {
      splashActive = true;
      splashStart = Date.now();
    }

    if (!inView && splashActive) {
      splashActive = false;
      splashTimeSent = (Date.now() - splashStart) / 1000;

      safe_gtag("splash_intro_time", {
        duration_seconds: splashTimeSent,
        event_category: "engagement"
      });
    }
  }

  window.addEventListener("scroll", checkSplash);

  // Also catch page exit
  document.addEventListener("visibilitychange", () => {
    if (document.visibilityState === "hidden" && splashActive) {
      const duration = (Date.now() - splashStart) / 1000;
      safe_gtag("splash_intro_time", {
        duration_seconds: duration,
        exited_without_scrolling: true,
        event_category: "engagement"
      });
      splashActive = false;
    }
  });


  // ================================================
  // C. CV downloads (CV button clicks)
  // ================================================

  // CV button
  document.querySelectorAll("[data-cv-download]").forEach(btn => {
    btn.addEventListener("click", () => {
      gtag("event", "cv_download", {
        event_category: "engagement",
        event_label: "CV button"
      });
    });
  });

  // JMP button
  document.querySelectorAll("[data-jmp-download]").forEach(btn => {
    btn.addEventListener("click", () => {
      gtag("event", "jmp_download", {
        event_category: "engagement",
        event_label: "JMP button"
      });
    });
  });

  // ================================================
  // D. Paper PDF downloads 
// ================================================
  const pdfLinks = document.querySelectorAll("[data-paper-download]");

  pdfLinks.forEach(function (link) {
    const paperId = link.dataset.paperDownload || "unknown";

    link.addEventListener("click", function () {
      safe_gtag("paper_download", {
        paper_id: paperId,
        event_category: "download",
        event_label: `PDF download: ${paperId}`
      });
    });
  });


  // =====================================================
  // G. Track clicks on VoxEU and Le Monde links
  // =====================================================
  document.querySelectorAll("[data-external-link]").forEach(btn => {
    btn.addEventListener("click", () => {
      safe_gtag("external_link_click", {
        event_category: "outbound",
        event_label: btn.dataset.externalLink
      });
    });
  });


    // =========================================
  // E. Track clicks on the hero micro-nav
  // =========================================
  document.querySelectorAll(".hero-mini-link").forEach(link => {
    link.addEventListener("click", () => {
      safe_gtag("mini_nav_click", {
        event_category: "navigation",
        event_label: link.textContent.trim()
      });
    });
  });


  // =====================================================
  // F. Track clicks + hovers on reference links
  // =====================================================
  document.querySelectorAll("[data-ref-name]").forEach(ref => {

    // Click tracking
    ref.addEventListener("click", () => {
      safe_gtag("reference_click", {
        event_category: "navigation",
        event_label: ref.dataset.refName
      });
    });

    // Hover tracking (fires once per session)
    let hovered = false;
    ref.addEventListener("mouseenter", () => {
      if (!hovered) {
        hovered = true;
        safe_gtag("reference_hover", {
          event_category: "engagement",
          event_label: ref.dataset.refName
        });
      }
    });

  });

});
